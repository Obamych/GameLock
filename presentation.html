# Обзор проекта

GameLock — Android-приложение на Kotlin + Jetpack Compose. Позволяет искать игры через RAWG API, добавлять их в личную библиотеку с рейтингом и статусом, смотреть детальную информацию (скриншоты, трейлеры, оценки Steam/Metacritic/RAWG).

🔍

Поиск

RAWG API + фильтры (жанры, магазины, сортировка). Страничная пагинация.

📚

Библиотека

Статусы: играю / прошёл / в планах. Пользовательский рейтинг, удаление.

📄

Деталка

Описание, скриншоты, трейлеры, оценки магазинов, теги, разработчики, Reddit.

👤

Профиль

Регистрация/вход (SHA-256). Привязка игр к пользователю. Статистика.

🗃️

Локально

Room SQLite. Композитный ключ (id игры + id пользователя). Flow.

🌐

API

RAWG, Steam, ML Kit Translation. Retrofit 2 + Gson.

Пароль — SHA-256 + соль

utils/PasswordHasher.kt
object PasswordHasher {
    fun hash(password: String): String {
        val salt = ByteArray(16).also { SecureRandom().nextBytes(it) }
        val digest = MessageDigest.getInstance("SHA-256")
        digest.update(salt)
        val hash = digest.digest(password.toByteArray(Charsets.UTF_8))
        return "${bytesToHex(salt)}:${bytesToHex(hash)}"
    }

    fun verify(password: String, stored: String): Boolean {
        val parts = stored.split(":")
        val salt = hexToBytes(parts[0])
        return hash(password) == stored  // повторно хеширует с той же солью
    }
}

Формат хранения: hex(salt):hex(hash). Никаких внешних библиотек.

class SearchViewModel(app: Application) : AndroidViewModel(app) {
    private val _uiState = MutableStateFlow<SearchUiState>(SearchUiState.Loading)
    val uiState: StateFlow<SearchUiState> = _uiState.asStateFlow()

    fun searchGames(query: String) {
        viewModelScope.launch {
            _uiState.value = SearchUiState.Loading
            repository.searchGames(query).fold(
                onSuccess = { _uiState.value = SearchUiState.Success(it) },
                onFailure = { _uiState.value = SearchUiState.Error(it.message ?: "Ошибка") }
            )
        }
    }
}
ui/screens/detail/DetailViewModel.kt
class DetailViewModel(app: Application) : AndroidViewModel(app) {

    private val prefs = PreferencesManager(app)
    private val userId: Int get() = prefs.currentUserId

    fun loadGame(gameId: Int) {
        viewModelScope.launch {
            _uiState.value = DetailUiState.Loading
            repository.getGameDetails(gameId, userId).fold(
                onSuccess = {
                    repository.updateLastAccessed(gameId, userId)
                    _uiState.value = DetailUiState.Success(it)
                },
                onFailure = { _uiState.value = DetailUiState.Error(it.message ?: "Ошибка") }
            )
        }
    }

    fun saveGameData(game: Game, status: GameStatus, rating: Float, review: String) {
        viewModelScope.launch {
            val updated = game.copy(
                userStatus = status, userRating = rating,
                userReview = review, lastAccessed = System.currentTimeMillis()
            )
            repository.saveGame(updated, userId)
            _uiState.value = DetailUiState.Success(updated)
        }
    }
}
ui/screens/library/LibraryViewModel.kt
class LibraryViewModel(app: Application) : AndroidViewModel(app) {

    // Flow из Room → реактивно обновляется при изменениях БД
    private val allGames: StateFlow<List<Game>> =
        repository.getLibraryGames(userId)
            .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())

    // Комбинируем фильтры: статус + поиск + сортировка
    val displayGames: StateFlow<List<Game>> = combine(
        allGames, _selectedStatus, _searchQuery, _sortType
    ) { games, status, query, sort ->
        var result = games
        if (status != null) result = result.filter { it.userStatus == status }
        if (query.isNotBlank()) result = result.filter { it.name.contains(query, ignoreCase = true) }
        when (sort) {
            SortType.NAME -> result.sortedBy { it.name }
            SortType.RATING -> result.sortedByDescending { it.overallRating }
            SortType.DATE_ADDED -> result.sortedByDescending { it.lastAccessed ?: 0L }
        }
    }.stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())
}

Generation guard — защита от гонок

Если пользователь быстро переключает страницы поиска, старый запрос не должен перезаписать новый результат:

private var loadGeneration = 0L

fun goToPage(page: Int) {
    loadGeneration++                     // увеличиваем счётчик
    val gen = loadGeneration            // запоминаем текущее значение
    viewModelScope.launch {
        val result = repository.searchGames(...)
        if (gen != loadGeneration) return@launch  // устарело — игнорируем
        _uiState.value = SearchUiState.Success(result)
    }
}

// ML Kit — определение языка + перевод
object TranslationService {
    suspend fun detectLanguage(text: String): SourceLang {
        val identifier = LanguageIdentification.getClient()
        val lang = identifier.identifyLanguage(text)
        return when (lang) {
            "en" -> SourceLang.ENGLISH
            "ru" -> SourceLang.RUSSIAN
            else -> SourceLang.OTHER
        }
    }

    suspend fun translate(text: String, source: SourceLang): Result<String> {
        val translator = Translation.getClient(
            TranslatorOptions.Builder()
                .setSourceLanguage(source.langCode)
                .setTargetLanguage("ru")
                .build()
        )
        translator.downloadModelIfNeeded().await()
        val result = translator.translate(text).await()
        translator.close()
        return Result.success(result)
    }
}
